/*-- Create and use the database
CREATE DATABASE IF NOT EXISTS football_league;
USE football_league;

-- Create Department table
CREATE TABLE IF NOT EXISTS Department (
    Dept_Code VARCHAR(10) PRIMARY KEY,
    Dept_short_name VARCHAR(50) NOT NULL,
    Dept_Head VARCHAR(100),
    Established_Year INT,
    Budget DECIMAL(12, 2)
);

-- Create Team table
CREATE TABLE IF NOT EXISTS Team (
    Team_ID VARCHAR(10) PRIMARY KEY,
    Dept_code VARCHAR(10),
    Manager VARCHAR(100),
    Name VARCHAR(100) NOT NULL,
    Player_Reg_ID VARCHAR(20),
    FOREIGN KEY (Dept_code) REFERENCES Department(Dept_Code)
);

-- Create Match table
CREATE TABLE IF NOT EXISTS `Match` (
    Match_ID VARCHAR(10) PRIMARY KEY,
    Stadium_Name VARCHAR(100),
    Date DATE,
    Team1_ID VARCHAR(10),
    Team2_ID VARCHAR(10),
    FOREIGN KEY (Team1_ID) REFERENCES Team(Team_ID),
    FOREIGN KEY (Team2_ID) REFERENCES Team(Team_ID)
);

-- Create Participate table (junction table between Department and Match)
CREATE TABLE IF NOT EXISTS Participate (
    Dept_code VARCHAR(10),
    Match_ID VARCHAR(10),
    PRIMARY KEY (Dept_code, Match_ID),
    FOREIGN KEY (Dept_code) REFERENCES Department(Dept_Code),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID)
);

-- Create Team_Match table (junction table between Team and Match)
CREATE TABLE IF NOT EXISTS Team_Match (
    Team_ID VARCHAR(10),
    Match_ID VARCHAR(10),
    PRIMARY KEY (Team_ID, Match_ID),
    FOREIGN KEY (Team_ID) REFERENCES Team(Team_ID),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID)
);

-- Create Player table
CREATE TABLE IF NOT EXISTS Player (
    Player_ID VARCHAR(10) PRIMARY KEY,
    Name VARCHAR(100) NOT NULL,
    Contact_No VARCHAR(15),
    Age INT,
    Position VARCHAR(50),
    Minutes_Played INT,
    Skills_Rating DECIMAL(3, 1),
    Team_ID VARCHAR(10),
    FOREIGN KEY (Team_ID) REFERENCES Team(Team_ID)
);

-- Create Match_Player table (junction table between Match and Player)
CREATE TABLE IF NOT EXISTS Match_Player (
    Match_ID VARCHAR(10),
    Player_ID VARCHAR(10),
    PRIMARY KEY (Match_ID, Player_ID),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID),
    FOREIGN KEY (Player_ID) REFERENCES Player(Player_ID)
);

-- Create Score table
CREATE TABLE IF NOT EXISTS Score (
    Score_ID VARCHAR(10) PRIMARY KEY,
    Number INT,
    Duration INT,
    Match_ID VARCHAR(10),
    Team_ID VARCHAR(10),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID),
    FOREIGN KEY (Team_ID) REFERENCES Team(Team_ID)
);

-- Create Contributed table (junction table between Player and Score)
CREATE TABLE IF NOT EXISTS Contributed (
    Player_ID VARCHAR(10),
    Score_ID VARCHAR(10),
    Match_ID VARCHAR(10),
    PRIMARY KEY (Player_ID, Score_ID),
    FOREIGN KEY (Player_ID) REFERENCES Player(Player_ID),
    FOREIGN KEY (Score_ID) REFERENCES Score(Score_ID),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID)
);

-- Create Referee table
CREATE TABLE IF NOT EXISTS Referee (
    Referee_ID VARCHAR(10) PRIMARY KEY,
    Role VARCHAR(50),
    Name VARCHAR(100) NOT NULL,
    Age INT,
    Institute VARCHAR(100)
);

-- Create Referee_Matches table (junction table between Referee and Match)
CREATE TABLE IF NOT EXISTS Referee_Matches (
    Referee_ID VARCHAR(10),
    Match_ID VARCHAR(10),
    PRIMARY KEY (Referee_ID, Match_ID),
    FOREIGN KEY (Referee_ID) REFERENCES Referee(Referee_ID),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID)
);

-- Create Punishment table
CREATE TABLE IF NOT EXISTS Punishment (
    Punishment_ID VARCHAR(10) PRIMARY KEY,
    Name VARCHAR(100) NOT NULL,
    Player_Semester VARCHAR(20),
    Referee_ID VARCHAR(10),
    FOREIGN KEY (Referee_ID) REFERENCES Referee(Referee_ID)
);

-- Create Match_Punishment table (junction table between Match and Punishment)
CREATE TABLE IF NOT EXISTS Match_Punishment (
    Match_ID VARCHAR(10),
    Punishment_ID VARCHAR(10),
    PRIMARY KEY (Match_ID, Punishment_ID),
    FOREIGN KEY (Match_ID) REFERENCES `Match`(Match_ID),
    FOREIGN KEY (Punishment_ID) REFERENCES Punishment(Punishment_ID)
);

-- Create Player_Punishment table (junction table between Player and Punishment)
CREATE TABLE IF NOT EXISTS Player_Punishment (
    Player_ID VARCHAR(10),
    Punishment_ID VARCHAR(10),
    PRIMARY KEY (Player_ID, Punishment_ID),
    FOREIGN KEY (Player_ID) REFERENCES Player(Player_ID),
    FOREIGN KEY (Punishment_ID) REFERENCES Punishment(Punishment_ID)
);

/*
-- SQL file with five SELECT queries for the Football League Database

-- 1. Simple SELECT-FROM-WHERE query (no joins, no subqueries, no aggregates)
-- Query to find all players who are forwards and have a skills rating above 8.0
SELECT Player_ID, Name, Age, Skills_Rating, Team_ID 
FROM Player
WHERE Position = 'Forward' AND Skills_Rating > 8.0;

-- 2. SELECT query using aggregate functions with GROUP BY
-- Query to find the total number of players in each team and their average skills rating
SELECT Team_ID, COUNT(*) AS NumberOfPlayers, AVG(Skills_Rating) AS AverageSkillsRating, 
       MAX(Skills_Rating) AS HighestSkillsRating
FROM Player
GROUP BY Team_ID
HAVING AVG(Skills_Rating) > 8.0;

-- 3. SELECT query using nested subqueries
-- Query to find all matches where the highest skilled player participated
SELECT m.Match_ID, m.Stadium_Name, m.Date, m.Team1_ID, m.Team2_ID
FROM `Match` m
WHERE m.Match_ID IN (
    SELECT mp.Match_ID
    FROM Match_Player mp
    WHERE mp.Player_ID = (
        SELECT p.Player_ID
        FROM Player p
        ORDER BY p.Skills_Rating DESC
        LIMIT 1
    )
);

-- 4. SELECT query using JOIN (First join query)
-- Query to find all punishments given in matches and the players who received them
SELECT m.Match_ID, m.Date, m.Stadium_Name, p.Name AS PlayerName, 
       pu.Name AS PunishmentType, pu.Player_Semester
FROM `Match` m
INNER JOIN Match_Punishment mp ON m.Match_ID = mp.Match_ID
INNER JOIN Punishment pu ON mp.Punishment_ID = pu.Punishment_ID
INNER JOIN Player_Punishment pp ON pu.Punishment_ID = pp.Punishment_ID
INNER JOIN Player p ON pp.Player_ID = p.Player_ID
ORDER BY m.Date, p.Name;

-- 5. SELECT query using JOIN (Second join query)
-- Query to find all matches with their scores, grouped by teams
SELECT t.Name AS TeamName, m.Match_ID, m.Date, m.Stadium_Name, 
       SUM(s.Number) AS TotalScore
FROM Team t
LEFT JOIN Team_Match tm ON t.Team_ID = tm.Team_ID
LEFT JOIN `Match` m ON tm.Match_ID = m.Match_ID
LEFT JOIN Score s ON (m.Match_ID = s.Match_ID AND t.Team_ID = s.Team_ID)
GROUP BY t.Name, m.Match_ID, m.Date, m.Stadium_Name
ORDER BY m.Date, t.Name;
