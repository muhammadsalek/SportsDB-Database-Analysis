/*
# Phase 5
#1. Retrieve All Departments
SELECT 
    Dept_Code, 
    Dept_Short_Name, 
    Dept_Head, 
    Established_Year, 
    Budget 
FROM Department;


# 2. Retrieve All Teams with Department Information
SELECT 
    T.Team_ID, 
    T.Name AS Team_Name, 
    T.Manager, 
    D.Dept_Short_Name AS Department, 
    D.Dept_Head AS Department_Head 
FROM Team T
JOIN Department D ON T.Dept_Code = D.Dept_Code;
3. Retrieve All Players with Their Team Information
SELECT 
    P.Player_ID, 
    P.Name AS Player_Name, 
    P.Contact_No, 
    P.Age, 
    P.Position, 
    P.Minutes_Played, 
    P.Skills_Rating, 
    T.Name AS Team_Name, 
    T.Manager AS Team_Manager 
FROM Player P
LEFT JOIN Team T ON P.Team_ID = T.Team_ID;


# 4. Retrieve All Matches with Participating Teams
SELECT 
    M.Match_ID, 
    M.Stadium_Name, 
    M.Date, 
    T1.Name AS Team_One, 
    T2.Name AS Team_Two 
FROM `Match` M
JOIN Team T1 ON M.Team1_ID = T1.Team_ID
JOIN Team T2 ON M.Team2_ID = T2.Team_ID;




#5. Retrieve Match Scores with Team Names
SELECT 
    S.Score_ID, 
    S.Number AS Goals, 
    S.Duration, 
    M.Match_ID, 
    M.Stadium_Name, 
    T.Name AS Team_Name 
FROM Score S
JOIN `Match` M ON S.Match_ID = M.Match_ID
JOIN Team T ON S.Team_ID = T.Team_ID;


# 6. Retrieve Players Who Contributed to Scores in Matches
SELECT 
    P.Player_ID, 
    P.Name AS Player_Name, 
    S.Number AS Goals_Scored, 
    M.Match_ID, 
    M.Stadium_Name, 
    M.Date 
FROM Contributed C
JOIN Player P ON C.Player_ID = P.Player_ID
JOIN Score S ON C.Score_ID = S.Score_ID
JOIN `Match` M ON C.Match_ID = M.Match_ID;


# 7. Retrieve Referees and Their Assigned Matches
SELECT 
    R.Referee_ID, 
    R.Name AS Referee_Name, 
    R.Role, 
    R.Institute, 
    M.Match_ID, 
    M.Stadium_Name, 
    M.Date 
FROM Referee_Matches RM
JOIN Referee R ON RM.Referee_ID = R.Referee_ID
JOIN `Match` M ON RM.Match_ID = M.Match_ID;


# 8. Retrieve Players Who Received Punishments with Details
SELECT 
    P.Player_ID, 
    P.Name AS Player_Name, 
    PN.Name AS Punishment_Type, 
    PN.Player_Semester, 
    R.Name AS Referee_Name, 
    R.Institute 
FROM Player_Punishment PP
JOIN Player P ON PP.Player_ID = P.Player_ID
JOIN Punishment PN ON PP.Punishment_ID = PN.Punishment_ID
JOIN Referee R ON PN.Referee_ID = R.Referee_ID;

# 9. Retrieve Matches and the Punishments Given in Each
SELECT 
    M.Match_ID, 
    M.Stadium_Name, 
    M.Date, 
    PN.Name AS Punishment_Type, 
    P.Name AS Player_Name, 
    R.Name AS Referee_Name 
FROM Match_Punishment MP
JOIN `Match` M ON MP.Match_ID = M.Match_ID
JOIN Punishment PN ON MP.Punishment_ID = PN.Punishment_ID
LEFT JOIN Player P ON PN.Player_ID = P.Player_ID
JOIN Referee R ON PN.Referee_ID = R.Referee_ID;


# 10. Retrieve Departments That Participated in Matches
SELECT 
    D.Dept_Code, 
    D.Dept_Short_Name AS Department, 
    M.Match_ID, 
    M.Stadium_Name, 
    M.Date 
FROM Participate P
JOIN Department D ON P.Dept_Code = D.Dept_Code
JOIN `Match` M ON P.Match_ID = M.Match_ID;
# 11. Retrieve All Teams with Their Players Count

SELECT 
    T.Team_ID, 
    T.Name AS Team_Name, 
    T.Manager, 
    COUNT(P.Player_ID) AS Number_of_Players 
FROM Team T
LEFT JOIN Player P ON T.Team_ID = P.Team_ID
GROUP BY T.Team_ID, T.Name, T.Manager;

# 12. Retrieve The Total Goals Scored By Each Team
SELECT 
    T.Team_ID, 
    T.Name AS Team_Name, 
    SUM(S.Number) AS Total_Goals 
FROM Score S
JOIN Team T ON S.Team_ID = T.Team_ID
GROUP BY T.Team_ID, T.Name
ORDER BY Total_Goals DESC;

# 13. Retrieve Referees Who Officiated the Most Matches
SELECT 
    R.Referee_ID, 
    R.Name AS Referee_Name, 
    R.Role, 
    COUNT(RM.Match_ID) AS Matches_Officiated 
FROM Referee_Matches RM
JOIN Referee R ON RM.Referee_ID = R.Referee_ID
GROUP BY R.Referee_ID, R.Name, R.Role
ORDER BY Matches_Officiated DESC;
# 14. Retrieve Players Sorted By Their Skill Rating
SELECT 
    Player_ID, 
    Name AS Player_Name, 
    Age, 
    Position, 
    Minutes_Played, 
    Skills_Rating 
FROM Player
ORDER BY Skills_Rating DESC;
# 15. Retrieve the Team With the Highest Average Player Skills
SELECT 
    T.Team_ID, 
    T.Name AS Team_Name, 
    ROUND(AVG(P.Skills_Rating), 2) AS Avg_Skill_Rating 
FROM Player P
JOIN Team T ON P.Team_ID = T.Team_ID
GROUP BY T.Team_ID, T.Name
ORDER BY Avg_Skill_Rating DESC
LIMIT 1;
# 16. Retrieve Matches Played at a Specific Stadium
SELECT 
    Match_ID, 
    Stadium_Name, 
    Date, 
    Team1_ID, 
    Team2_ID 
FROM `Match`
WHERE Stadium_Name = 'Lusail Stadium';
